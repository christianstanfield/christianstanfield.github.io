Functions

Syntax

var functionName = function(parameter, parameter) { body; }

Example

var sayHello = function() { return 'hello'; } sayHello(); // => 'hello'

Alternate Example

var add = function(x, y) { return x + y; } add(1,1); // => 2

Objects

Syntax

var objectName = new Object(); var objectName = {};

Example

var person = { firstName: 'Indiana', lastName: 'Jones', age: 50 }; person.firstName; // => 'Indiana' person['firstName']; // => 'Indiana'

Add or Update Properties

person.firstName = 'Indy'; person.career = 'archeologist'; person; // => { firstName: 'Indy', lastName: 'Jones', age: 50, career: 'archeologist' }

Nested Objects

person.accessories = { headGear: 'hat', weapon: 'whip' }; person.accessories.weapon; // => 'whip'

Methods

Syntax

object.methodName = function(parameter, parameter) { body; }

Example

person.setWeapon = function(newWeapon) {this.accessories.weapon = newWeapon;} person.setWeapon('bazooka'); person.accessories.weapon; // => 'bazooka'

Chaining Methods

var myString = 'I like programming'; myString.concat(' and sushi!').toUpperCase(); // => 'I LIKE PROGRAMMING AND SUSHI!'

Constructors

Declaring

function Robot(name, type) { this.name = name; this.type = type; this.material = 'metal'; this.newType = function(newType) { this.type = newType; } var loveInterest = 'EVE'; }

Creating an Instance

var wallE = new Robot('WALL-E', 'Waste Allocation Load Lifter Earth-Class'); wallE.name; // => 'WALL-E' wallE.material; // => 'metal' wallE.newType('movieStar'); // wallE.type; => 'movieStar' wallE.loveInterest; // => private parameter

Loops

While / Do

var n = 1; while (n <= 5) { console.log('n is equal to ' + n); n++; } // => 'n is equal to 1', 'n is equal to 2', etc var n = 1; do { console.log('n is equal to ' + n); n++; } while (n < 1); // => 'n is equal to 1' (execute code, then check while condition)

For / In

var tacos = ['asada', 'carnitas', 'pollo', 'al pastor']; for (var i = 0; i < tacos.length; i++) { console.log( 'I would like to order a ' + tacos[i] + ' taco.'); } // => 'I would like to order a asada taco.', etc for (i in tacos) { console.log( 'Seriously I really would like a ' + tacos[i] + ' taco.'); } // => 'Seriously I really would like a asada taco.', etc

Switch / Case

var yourChoice = prompt('What can I do for you?'); switch (yourChoice) { case 'money': console.log('This window is for deposits only!'); break; case 'more money': console.log('Seriously, not gonna happen.'); break; case 'women': console.log('How about a taco?'); break; default: console.log('You keep using that word. I'm not sure it means what you think it means.'); }

Conditionals

If, Else if, Else

if (name == 'Inigo Montoya' && killedFather == true) { console.log('Prepare to die!'); } else if (name == 'Inigo Montoya' && killedFather == false) { console.log('Sorry to have bothered you.'); } else { console.log('Want to get tacos?'); }
